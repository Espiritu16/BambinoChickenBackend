  CREATE OR REPLACE EDITIONABLE TRIGGER "BAMBINO"."TRG_CAJA_AUDITORIA"
      BEFORE INSERT OR UPDATE ON BAMBINO.CAJA
      FOR EACH ROW
      BEGIN
	IF INSERTING THEN
	  IF :NEW.CREADO_EN IS NULL THEN :NEW.CREADO_EN := SYSTIMESTAMP; END IF;
	  IF :NEW.CREADO_POR IS NULL THEN :NEW.CREADO_POR := SYS_CONTEXT('USERENV','SESSION_USER'); END IF;
	END IF;
	:NEW.ACTUALIZADO_EN := SYSTIMESTAMP;
	IF :NEW.ACTUALIZADO_POR IS NULL THEN
	  :NEW.ACTUALIZADO_POR := SYS_CONTEXT('USERENV','SESSION_USER');
	END IF;
      END;
ALTER TRIGGER "BAMBINO"."TRG_CAJA_AUDITORIA" ENABLE
